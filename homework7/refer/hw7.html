自变量的含义：
· creded(借贷比率)：用户所有行为中，采用贷记卡（或称为信用卡）交易的次数占所有采用贷记卡或者借记卡（或称为储蓄卡）交易次数的比率；
· debitF(借记类F)：用户使用储蓄卡的消费频率；
· meanpay(所有行为均值)：用户被记录的所有交易行为的平均金额；
· billnum(交易笔数)：用户自从第一笔交易记录开始，被记录的交易总笔数；
· debitM(借记类M)：用户使用储蓄卡所有消费金额的平均值；
· zhongxingM(中型银行M)：用户使用中型银行卡所有消费金额的平均值；
· sidaM(四大行M)：用户使用四大行银行卡所有消费金额的平均值；
· xindaiR(信贷类R)：用户上一次小额贷款的时间到现在的距离；
· cardnum(银行卡数)：用户银行卡的数量；
· xindaiF(信贷类F)：用户小型贷款的频率；
· maxpay(所有行为最大值)：用户自从第一笔交易开始，所有交易行为的金额的最大值；
· zhuanzhangF(转账类F)：用户转账的频率；
· gongjiaoF(公缴类F)：用户交水，电，煤气费等交易行为的频率；
· zhuanzhangR(转账类R)：用户最近一次转账的时间到现在的距离；
· age(年龄)：用户的年龄；
· gongjiaoR(公缴类R)：用户上一次交水，电，煤气费等交易的时间到现在的距离；
· sidaF(四大行F)：用户使用四大行银行卡的频率;
· sidaR(四大行R): 用户上一次使用四大行银行卡的时间到现在的距离;
· zhongxingF(中型银行F)：用户使用中型银行卡的频率；
· zhongxingR(中型银行R)：用户上一次使用中型银行卡的时间到现在的距离;
· xiaofeiF(消费类F)：用户日常消费行为的频率；
· jinkaF(金卡类F)：用户使用金卡消费的频率；
· zhuanzhangM(转账类M)：用户转账金额的平均值；
· gongjiaoM(公缴类M)：用户交水，电，煤气费等交易的平均金额；
· youxiM(游戏类M)：用户购买游戏点卡的金额的平均值；
· xindaiS(信贷类S)：用户的小额贷款的标准差；

2.按照7:3划分训练集和测试集，用多种机器学习模型进行建模；在测试集上进行测试，绘制ROC曲线，计算AUC值

library(pROC) #导入pROC包
set.seed(1234)
dat0[-27] <- scale(dat0[-27]) # 对所有自变量进行标准化
index <- sample(1:nrow(dat0), round(nrow(dat0)*7/10)) # 确定训练集数据下标
dat.train <- dat0[index,] #训练集
dat.test <- dat0[-index,] #测试集

# 逻辑回归模型
logis <- glm(formula = black ~ ., family = binomial(link = logit), data = dat.train)  #建立模型
logis_pred <- predict.glm(logis, type = "response", newdata = dat.test) # 进行预测
logis.roc <- roc(dat.test$black, logis_pred)
plot(logis.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="逻辑回归模型的ROC和AUC", xlab = "FPR", ylab = "TPR")  #画出ROC并计算AUC

# kNN模型
library("kknn")

knn <- kknn(formula = black ~ ., dat.train,dat.test, k=7, distance = 2) # 建立模型
knn_pred <- fitted(knn) #进行预测
knn.roc <- roc(dat.test$black, knn_pred)
plot(knn.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="kNN模型的ROC和AUC", xlab = "FPR", ylab = "TPR")  #画出ROC并计算AUC

# 决策树模型
library("rpart")
dtree <- rpart(black~., data = dat.train, method = "class") #建立模型
dtree_pred <- predict(dtree, newdata = dat.test, type = "class")  #进行预测
dtree_pred <- as.numeric(dtree_pred)-1
dtree.roc <- roc(dat.test$black, dtree_pred) 
plot(dtree.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="决策树模型的ROC和AUC", xlab = "FPR", ylab = "TPR")  #画出ROC并计算AUC

# boosting模型
