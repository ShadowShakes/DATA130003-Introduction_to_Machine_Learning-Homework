library(pROC) #pROC
set.seed(1234)
dat0[-27] <- scale(dat0[-27]) # 
index <- sample(1:nrow(dat0), round(nrow(dat0)*7/10)) # 
dat.train <- dat0[index,] #
dat.test <- dat0[-index,] #

# 
logis <- glm(formula = black ~ ., family = binomial(link = logit), data = dat.train)  #
logis_pred <- predict.glm(logis, type = "response", newdata = dat.test) # 
logis.roc <- roc(dat.test$black, logis_pred)
plot(logis.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="ROCAUC", xlab = "FPR", ylab = "TPR")  #ROCAUC

# kNN
library("kknn")

knn <- kknn(formula = black ~ ., dat.train,dat.test, k=7, distance = 2) # 
knn_pred <- fitted(knn) #
knn.roc <- roc(dat.test$black, knn_pred)
plot(knn.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="kNNROCAUC", xlab = "FPR", ylab = "TPR")  #ROCAUC

# 
library("rpart")
dtree <- rpart(black~., data = dat.train, method = "class") #
dtree_pred <- predict(dtree, newdata = dat.test, type = "class")  #
dtree_pred <- as.numeric(dtree_pred)-1
dtree.roc <- roc(dat.test$black, dtree_pred) 
plot(dtree.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="ROCAUC", xlab = "FPR", ylab = "TPR")  #ROCAUC

# boosting
library("adabag")
dat.train$black <- as.factor(dat.train$black)
adabst <- boosting(black~., data = dat.train) #
adabst_pred <- predict(adabst, dat.test)  #
adabst_pred <- as.numeric(adabst_pred$class)
adabst.roc <- roc(dat.test$black, adabst_pred)
plot(adabst.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="adaboostROCAUC", xlab = "FPR", ylab = "TPR")  #ROCAUC

# 
library(randomForest)
ranfor <- randomForest(black~., data = dat.train) #
ranfor_pred <- predict(ranfor, dat.test)  #
ranfor_pred <- as.numeric(ranfor_pred)-1
ranfor.roc <- roc(dat.test$black, ranfor_pred)
plot(ranfor.roc, print.auc=TRUE, auc.polygon=TRUE, legacy.axes=TRUE, grid=c(0.1, 0.2), grid.col=c("green", "red"), max.auc.polygon=TRUE,  auc.polygon.col="skyblue", main="ROCAUC", xlab = "FPR", ylab = "TPR")  #ROCAUC
